\begin{definition}

  Functors map between categories while respecting composition and identity.

  $$ F : C \rtarw D $$
  $$ F : obj(C) \rtarw obj(D) $$
  $$ F : hom(C) \rtarw hom(D) $$

\end{definition}

\begin{tikzcd}
  A \arrow[r, "f"] & B \\
  F(A) \arrow[r, "F(f)"] & F(B) \\
\end{tikzcd}
